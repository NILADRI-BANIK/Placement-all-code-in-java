Automorphic Number:-
import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int n=sc.nextInt();
		int sq=n*n;
		int d=0;
		int t=n;
		while(t>0){
		    d++;
		    t=t/10;
		}
		int m=(int)Math.pow(10,d);
		if(sq%m == n){
		    System.out.print("Automorphic Number");
		}else{
		    System.out.print("not Automorphic Number");
		}
	}
}
